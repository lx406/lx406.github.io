# Symbol

Symbol的出现是为了解决对象中属性命名的重复，Symbol保证了属性名称的唯一性。

Symbol也成为了JavaScript的第七种数据类型。

这样，对象的属性名可以有两种数据类型Symbol和字符串。

let s = Symbol();
typeof s // 'symbol'

// s是独一无二的

注意：Symbol不是对象，声明时不能用new，它是一种类似于字符串的类型。

let a = Symbol('foo');
let b = Symbol('bar');

// 上面两个变量均为Symbol类型，如果不加上参数，在控制台打出的都是Symbol(),无法区分，所以加上变量即可区分

注：参数只是为了区分Symbol，如果同一个参数，他们的值是不同的。

// Symbol不能与其他类型运算，但可以显示转化为字符串。

let a = Symbol('foo');
a.toString();// "Symbol(foo)"

// Symbol也可以转为布尔值，但不能转化为数值

!a // false
a + 2; // TypeError

# 作为属性名的Symbol

// Symbol类型的值都是唯一的，所以可以避免对象中属性重复的问题。

// 对象中增加Symbol类型的方式
// 第一种
let a = {};
let mySymbol = Symbol();
a[mySymbol] = 'Hello';

// 第二种
let a = {
    [mySymbol]: 'Hello'
}

// 第三种
let a = {};
Object.defineProperty(a, mySymbol, {
    value: 'Hello'
})

注：

  1. Symbol无法使用点运算符调用（因为点运算符后总跟着字符串）。
  2. 在对象中，Symbol类型必须使用[]定义。
  3. Symbol类型作属性名时还是公开属性，并非私有属性。
// 常见用法：定义常量
const COLOR_RED    = Symbol();
const COLOR_GREEN  = Symbol();

function getComplement(color) {
  switch (color) {
    case COLOR_RED:
      return COLOR_GREEN;
    case COLOR_GREEN:
      return COLOR_RED;
    default:
      throw new Error('Undefined color');
    }
}

console.log(getComplement(COLOR_RED)); // Symbol()
console.log(getComplement(COLOR_YELLOW)); // error

注：Symbol与switch可以很好的搭配，在常量中属性的值是什么往往并不重要，只要保证属性名不同即可，所以Symbol很好的解决了这个问题

最常见的场景：消除魔法字符串。

魔法字符串：在代码之中多次出现、与代码形成强耦合的某一个具体的字符串或者数值。

function getArea(shape, options) {
  var area = 0;

  switch (shape) {
    case 'Triangle': // 魔术字符串
      area = .5 * options.width * options.height;
      break;
    /* ... more code ... */
  }

  return area;
}

getArea('Triangle', { width: 100, height: 100 });

// 上面代码中 "Triangle"即为魔法字符串，改为下面优化代码。
const shapeType = {
    triangle:Symbol()
}
function getArea(shape, options) {
  var area = 0;

  switch (shape) {
    case shape.triangle: // 魔术字符串
      area = .5 * options.width * options.height;
      break;
    /* ... more code ... */
  }

  return area;
}

getArea(shape.triangle, { width: 100, height: 100 });

# 属性名的遍历

平时使用的遍历方法，如：Object.Keys(),for...of,for...in等，都不能将Symbol类型的值遍历出来。

  1. 使用 ==Object.getOwnPropertySymbols(obj)== 获取
let mySymbol = Symbol('foo');
let a = {
    [mySymbol]:'hello'
}
Object.getOwnPropertySymbols(a); // [Symbol(foo)]
  1. 使用==Reflect.ownKeys(obj)==获取
let mySymbol = Symbol('foo');
let a = {
    [mySymbol]:'hello'
    mySymbol:'hello'
}
Reflect.ownKeys(a); // ['mySymbol',Symbol(foo)]

注：我们可以借助Symbol类型无法使用常见方法遍历的特性，造成一种非私有的内部方法的效果

# Symbol.for(),Symbol.keyFor()

  1. Symbol.for() 这个方法先搜索可以搜索到的Symbol变量，比较定义时传入的参数，如果参数一致，则把之前的值赋给它，如果没搜索到，则可以新定义一个Symbol类型的变量。
let a = Symbol.for('foo');
let b = Symbol.for('foo');

a === b; //true

// 使用Symbol.for()创建的变量会被登记在当前作用于下供搜索
  1. Symbol.keyFor() Symbol.keyFor方法返回一个已登记的 Symbol 类型值的key。
let a = Symbol.for('foo');
Symbol.keyFor(a); // "foo"

let b = Symbol('bar');
Symbol.keyFor(b); // undefined

# 内置的Symbol值

  1. Symbol.hasInstance
// 当使用instanceof时会调用对象内部的Symbol.hasInstance方法

let b = {
    [Symbol.hasInstance](obj) {
        return obj instanceof Array
    }
}

console.log([] instanceof b); // true
  1. Symbol.isConcatSpreadable
// 数组默认可以展开，伪数组默认无法展开
let a = ['a','b'];
['c'].concat(a,'d');// ['c','a','b','d']

let b = ['e','f'];
b[Symbol.isConcatSpreadable] = false;
['c'].concat(b, 'd');// ['c',['e','f'],'d']
  1. Symbol.species
class arr extends Array{
    constructor(){
        super();
    }
    static get [Symbol.species](){
        return Array;
    }
}

var a = new arr(1,2,3);
var mapped = a.map(x => x * x);
console.log(mapped instanceof arr); // false
console.log(mapped instanceof Array); // true
  1. Symbol.interator

对象的Symbol.iterator属性，指向该对象的默认遍历器方法。

对象进行for...of循环时，会调用Symbol.iterator方法，返回该对象的默认遍历器
