# 冒泡排序

/**
 * 冒泡排序
 * 两两交换，直到遍历n遍
 * 输入[5,8,6,2,3,1,4,7]
 * 输出[1,2,3,4,5,6,7,8]
 * 稳定排序（相同的两个元素位置不变）
 * 时间复杂度O(n2)
 * 平均空间复杂度O(1)
 */

function bubbleSort(arr) {
  let j = 0;
  while (j < arr.length) {
    for (let i = 0; i < arr.length - 1; i++) {
      if (arr[i] > arr[i + 1]) {
        [arr[i], arr[i + 1]] = [arr[i + 1], arr[i]];
      }
    }
    j++
  }
}

let arr1 = [5, 8, 6, 2, 3, 1, 4, 7];
bubbleSort(arr1);
console.log(arr1);

# 插入排序

/**
 * 插入排序
 * 将数组分成有序和无序两部分，首先将左一当做有序，将无序的第一个元素向有序数组中插入，直到全部插入结束
 * 输入[5,8,6,2,3,1,4,7]
 * 输出[1,2,3,4,5,6,7,8]
 * 稳定排序（相同的两个元素位置不变）
 * 时间复杂度 O(n2)
 * 空间复杂度
 */

function insertSort(arr) {
  let i = 1;
  while (i < arr.length) {
    for (let j = i; j > 0; j--) {
      if (arr[j] < arr[j - 1]) {
        [arr[j], arr[j - 1]] = [arr[j - 1], arr[j]]
      } else {
        break
      }
    }
    i++
  }
}

let arr2 = [5, 8, 6, 2, 3, 1, 4, 7]
insertSort(arr2)
console.log(arr2)

# 归并排序

/**
 * 归并排序
 * 先将数组两两分组（sort）直到不能再分，通过递归的方式，将分好的组进行排序（merge）
 * 输入[5,8,6,2,3,1,4,7]
 * 输出[1,2,3,4,5,6,7,8]
 * 稳定排序（相同的两个元素位置不变）
 * 时间复杂度 O(nlogn)
 * 空间复杂度
 */

function mergeSort(arr) {
  let temp = []
  sort(arr, 0, arr.length - 1, temp)
}

function sort(arr, left, right, temp) {
  if (left < right) {
    let mid = parseInt((left + right) / 2)
    sort(arr, left, mid, temp)
    sort(arr, mid + 1, right, temp)
    merge(arr, left, mid, right, temp)
  }
}

function merge(arr, left, mid, right, temp) {
  let leftIndex = left
  let rightIndex = mid + 1
  let t = 0
  while (leftIndex < mid + 1 && rightIndex <= right) {
    if (arr[leftIndex] <= arr[rightIndex]) {
      temp[t++] = arr[leftIndex]
      leftIndex++
    } else {
      temp[t++] = arr[rightIndex]
      rightIndex++
    }
  }
  while (leftIndex <= mid) {
    temp[t++] = arr[leftIndex]
    leftIndex++
  }
  while (rightIndex <= right) {
    temp[t++] = arr[rightIndex]
    rightIndex++
  }
  t = 0;
  //将temp中的元素全部拷贝到原数组中
  while (left <= right) {
    arr[left++] = temp[t++];
  }
}

let arr2 = [5, 8, 6, 2, 3, 1, 4, 7]
mergeSort(arr2)
console.log(arr2)
