# 选择排序

/**
 * 直接选择排序
 * 从无序的数组中取出最小的数与第一个元素交换
 * 不稳定排序
 * 输入[5,8,6,2,3,1,4,7]
 * 输出[1,2,3,4,5,6,7,8]
 * 时间复杂度O(n2)
 * 平均空间复杂度O
 */

function selectionSort(arr) {
  let j = 0;
  while (j < arr.length) {
    let min = arr[j]
    let index = j
    for (let i = j + 1; i < arr.length; i++) {
      if (arr[i] < min) {
        min = arr[i]
        index = i
      }
    }
    [arr[j], arr[index]] = [arr[index], arr[j]]
    j++
  }
}

let arr2 = [5, 8, 6, 2, 3, 1, 4, 7]
selectionSort(arr2)
console.log(arr2)

# 快速排序

/**
 * 快速排序
 * 选择其中一个元素作为基数，并从两侧向里遍历，大于元素则放在元素右侧，反之放到左侧，并递归直至长度为1
 * 不稳定排序
 * 输入[5,8,6,2,3,1,4,7]
 * 输出[1,2,3,4,5,6,7,8]
 * 时间复杂度O(nlogn)
 */

function quickSort(arr) {
  sort(arr, 0, arr.length - 1)
}

function sort(arr, left, right) {
  let baseNum = arr[left] // 基数
  let leftIndex = left // 左指针
  let rightIndex = right // 右指针
  if (left < right) {
    while (leftIndex !== rightIndex) {
      while (leftIndex < rightIndex && arr[rightIndex] > baseNum) {
        rightIndex--
      }
      arr[leftIndex] = arr[rightIndex]
      while (leftIndex < rightIndex && arr[leftIndex] < baseNum) {
        leftIndex++
      }
      arr[rightIndex] = arr[leftIndex]
    }
    arr[leftIndex] = baseNum
    sort(arr, left, leftIndex - 1)
    sort(arr, leftIndex + 1, right)
  }
}

let arr2 = [5, 8, 6, 2, 3, 1, 4, 7]
quickSort(arr2)
console.log(arr2)

# 希尔排序

/**
 * 希尔排序
 * 从大间隔向小间隔逐次遍历
 * 不稳定排序
 * 输入[5,8,6,2,3,1,4,7]
 * 输出[1,2,3,4,5,6,7,8]
 * 时间复杂度（nlog2n）
 */

// 第一遍：gap = 4 ; 3,1,4,2,5,8,6,7
// 第二遍：gap = 2 ; 3,1,4,2,5,8,6,7
// 第三遍: gap = 1 ; 1,2,3,4,5,6,7,8

function shellSort(arr) {
  // 第一步找间隔
  let gap = parseInt(arr.length / 2)
  // 选定gap
  for (gap; gap > 0; gap = parseInt(gap / 2)) {
    // 从gap个元素开始比较
    for (let i = gap; i < arr.length; i++) {
      let j = i
      let current = arr[i]
      while (j - gap >= 0 && arr[j - gap] > current) {
        arr[j] = arr[j-gap]
        j = j - gap 
      }
      arr[j] = current
    }
  }
}

let arr2 = [5, 8, 6, 2, 3, 1, 4, 7]
shellSort(arr2)
console.log(arr2)
